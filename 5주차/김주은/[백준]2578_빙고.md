### 문제풀이 핵심아이디어 또는 새롭게 알게된 내용: 
    핵심 아이디어: 가로줄, 세로줄, 왼쪽 대각선, 오른쪽 대각선인 경우 나눠서 구하기 


### 코드의 시간복잡도와 그 이유:
    O(1)
    이유: 상수 시간 동안 실행
    
    
### 주석이 가득 담긴 코드:
```python
import sys
input = sys.stdin.readline

cs = [list(map(int, input().split())) for _ in range(5)] # 빙고판에 쓰여진 수
mc = [] # 사회자가 부르는 수
for _ in range(5):
    mc += list(map(int, input().split()))

# 빙고 게임
def bingo_check():
    bingo = 0 # 빙고의 개수

    # 가로줄
    for x in cs:
        if x.count(0) == 5: # 가로줄 위에 있는 5개의 모든 수가 지워지는 경우
            bingo += 1 # 빙고 추가

    # 세로줄
    for i in range(5): 
        y = 0
        for j in range(5):
            if cs[j][i] == 0:
                y += 1
        if y == 5: # 세로줄 위에 있는 5개의 모든 수가 지워지는 경우
            bingo += 1 # 빙고 추가

    # 왼쪽 대각선
    tmp1 = 0 # 왼쪽 대각선 위에 사회자가 부른 수와 일치하는 수의 개수 
    for i in range(5):
        if cs[i][i] == 0: # 왼쪽 대각선의 수가 사회자가 부른 수와 일치하는  경우 
            tmp1 += 1 # 왼쪽 대각선 위의 숫자가 사회자가 부른 숫자가 일치하는 개수 추가
    if tmp1 == 5: # 왼쪽 대각선 위에 있는 5개의 모든 수가 지워지는 경우
        bingo += 1 # 빙고 추가 

    # 오른쪽 대각선
    tmp2 = 0 # 오른쪽 대각선 위에 사회자가 부른 수와 일치하는 수의 개수
    for i in range(5):
        if cs[i][4 - i] == 0: # 오른쪽 대각선의 수가 사회자가 부른 수와 일치하는  경우
            tmp2 += 1 # 오른쪽 대각선 위의 숫자가 사회자가 부른 숫자가 일치하는 개수 추가
    if tmp2 == 5: # 오른쪽 대각선 위에 있는 5개의 모든 수가 지워지는 경우
        bingo += 1 # 빙고 추가

    return bingo


cnt = 0 # 사회자가 숫자를 부르는 횟수
for i in range(25):
    for x in range(5):
        for y in range(5):
            if mc[i] == cs[x][y] : # 사회자가 부르는 수가 빙고판에 있는 수와 같은 경우
                cs[x][y] = 0 # 해당하는 숫자를 0으로 만듦
                cnt += 1 # 횟수 더함

    if cnt >= 12: # 빙고가 3개 나오기 위해 최소 12번을 말해야 함
        result = bingo_check()
        if result >= 3: # 빙고가 세 개 이상인 경우
            print(i + 1) # 사회자가 몇 번쩨 수를 부른 후 철수가 "빙고"를 외치게 되는지 출력
            break

```
